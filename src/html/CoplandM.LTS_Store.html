States


Inductive St: Set :=
| stop: St
| conf: AnnoTerm → St
| rem: St → St
| ls: St → AnnoTerm → St
| bsl: nat → St → AnnoTerm → St
| bsr: nat → St → St
| bp: nat → St → St → St
| bpl: St → St
| bpr: St → St.


Labeled Transition System

The label in a transition is either an event or None when the transition is silent. Notice the use of annotations to provide the correct number for each event.

Inductive step: St → option PutPoll → St → Prop :=
Measurement

| stasp:
    ∀ r x,
      step (conf (aasp r x))
           (Some (none_putpoll (fst r)))
           (stop)
Remote call

| statt:
    ∀ r x q,
      step (conf (aatt r q x))
           (Some (putpoll (fst r) (fst r) (snd r)))
           (rem (conf x))
| stattstep:
    ∀ st0 st1 ev,
      step st0 ev st1 →
      step (rem st0) ev (rem st1)
| stattstop:
   
      step (rem (stop))
           None
           (stop)
Linear Sequential Composition

| stlseq:
    ∀ r x y,
      step (conf (alseq r x y))
           None
           (ls (conf x) y)
| stlseqstep:
    ∀ st0 ev st1 t,
      step st0 ev st1 →
      step (ls st0 t) ev (ls st1 t)
| stlseqstop:
    ∀ t,
      step (ls (stop) t) None (conf t)
Branching Sequential Composition

| stbseq:
    ∀ r s x y,
      step (conf (abseq r s x y))
           None
           (bsl (snd r) (conf x)
                y)
| stbslstep:
    ∀ st0 ev st1 j t,
      step st0 ev st1 →
      step (bsl j st0 t) ev (bsl j st1 t)
| stbslstop:
    ∀ j t,
      step (bsl j (stop) t)
           None
           (bsr j (conf t))
| stbsrstep:
    ∀ st0 ev st1 j,
      step st0 ev st1 →
      step (bsr j st0) ev (bsr j st1)
| stbsrstop:
    ∀ j,
      step (bsr j (stop))
           None
           (stop)

Branching Parallel composition

| stbpar:
    ∀ r s x y,
      step (conf (abpar r s x y))
           
           None
           (bp (snd r)
               (bpl (conf x))
               (bpr (conf y)))
| stbparl: ∀ x,
    step (bpl (conf x))
         (Some (putpoll (fst (range x)) (fst (range x)) (snd (range x) - 1)))
         (conf x)
| stbparr: ∀ y,
    step (bpr (conf y))
         (Some (putpoll (fst (range y)) (fst (range y)) (snd (range y) - 1)))
         (conf y)
                             
| stbpstepleft:
    ∀ st0 st1 st2 ev j,
      step st0 ev st2 →
      step (bp j st0 st1) ev (bp j st2 st1)
| stbpstepright:
    ∀ st0 st1 st2 ev j,
      step st1 ev st2 →
      step (bp j st0 st1) ev (bp j st0 st2)
| stbpstop:
    ∀ j,
      step (bp j (stop) (stop))
           None
           (stop).
Hint Constructors step : core.


Transitive Closures


Inductive lstar: St → list PutPoll → St → Prop :=
| lstar_refl: ∀ st, lstar st [] st
| lstar_tran: ∀ st0 e st1 tr st2,
    step st0 (Some e) st1 → lstar st1 tr st2 → lstar st0 (e :: tr) st2
| lstar_silent_tran: ∀ st0 st1 tr st2,
    step st0 None st1 → lstar st1 tr st2 → lstar st0 tr st2.
Hint Resolve lstar_refl : core.

Lemma lstar_transitive:
  ∀ st0 tr0 st1 tr1 st2,
    lstar st0 tr0 st1 →
    lstar st1 tr1 st2 →
    lstar st0 (tr0 ++ tr1) st2.

Transitive closure without labels.

Inductive star: St → St → Prop :=
| star_refl: ∀ st, star st st
| star_tran: ∀ st0 e st1 st2,
    step st0 e st1 → star st1 st2 → star st0 st2.
Hint Resolve star_refl : core.

Lemma star_transitive:
  ∀ st0 st1 st2,
    star st0 st1 →
    star st1 st2 →
    star st0 st2.

Lemma lstar_star:
  ∀ st0 tr st1,
    lstar st0 tr st1 → star st0 st1.

Lemma star_lstar:
  ∀ st0 st1,
    star st0 st1 → ∃ tr, lstar st0 tr st1.


Correct Path Exists


Lemma star_strem:
  ∀ st0 st1,
    star st0 st1 → star (rem st0) (rem st1).

Lemma star_stls:
  ∀ st0 st1 t,
    star st0 st1 → star (ls st0 t) (ls st1 t).

Lemma star_stbsl:
  ∀ st0 st1 j t,
    star st0 st1 →
    star (bsl j st0 t) (bsl j st1 t).

Lemma star_stbsr:
  ∀ st0 st1 j,
    star st0 st1 →
    star (bsr j st0) (bsr j st1).

Lemma lstar_stls :
  ∀ st0 st1 t tr,
    lstar st0 tr st1 → lstar (ls st0 t) tr (ls st1 t).

Lemma lstar_strem : ∀ st st' tr,
    lstar st tr
          st' →
    lstar (rem st) tr (rem st').

Lemma lstar_stbsl:
  ∀ st0 st1 j t tr,
    lstar st0 tr st1 →
    lstar (bsl j st0 t) tr (bsl j st1 t).

Lemma lstar_stbsr:
  ∀ st0 st1 j tr,
    lstar st0 tr st1 →
    lstar (bsr j st0) tr (bsr j st1).

Lemma star_stbp:
  ∀ st0 st1 st2 st3 j,
    star st0 st1 →
    star st2 st3 →
    star (bp j st0 st2) (bp j st1 st3).

Theorem correct_path_exists:
  ∀ t,
    star (conf t) (stop).

Progress


Definition halt st :=
  match st with
  | stop ⇒ True
  | _ ⇒ False
  end.


Termination


Inductive nstar: nat → St → St → Prop :=
| nstar_refl: ∀ st, nstar 0 st st
| nstar_tran: ∀ st0 st1 st2 e n,
    nstar n st0 st1 → step st1 e st2 → nstar (S n) st0 st2.
Hint Resolve nstar_refl : core.

Lemma nstar_transitive:
  ∀ m n st0 st1 st2,
    nstar m st0 st1 →
    nstar n st1 st2 →
    nstar (m + n) st0 st2.

Lemma nstar_star:
  ∀ n st0 st1,
    nstar n st0 st1 → star st0 st1.

Lemma star_nstar:
  ∀ st0 st1,
    star st0 st1 →
    ∃ n, nstar n st0 st1.

Size of a term (number of steps to reduce).

Fixpoint tsize t: nat :=
  match t with
  | aasp _ _ ⇒ 1
  | aatt _ _ x ⇒ 2 + tsize x
  | alseq _ x y ⇒ 2 + tsize x + tsize y
  | abseq _ _ x y ⇒ 3 + tsize x + tsize y
  | abpar _ _ x y ⇒ 4 + tsize x + tsize y
  end.

Size of a state (number of steps to reduce).

Fixpoint ssize s: nat :=
  match s with
  | stop ⇒ 0
  | conf t ⇒ tsize t
  | rem x ⇒ 1 + ssize x
  | ls x t ⇒ 1 + ssize x + tsize t
  | bsl _ x t ⇒ 2 + ssize x + tsize t
  | bsr _ x ⇒ 1 + ssize x
  | bp _ x y ⇒ 1 + ssize x + ssize y
  | bpl x ⇒ 1 + ssize x
  | bpr y ⇒ 1 + ssize y
  end.

Halt state has size 0.

Lemma halt_size:
  ∀ st,
    halt st ↔ ssize st = 0.

A state decreases its size by one.

Lemma step_size:
  ∀ st0 e st1,
    step st0 e st1 →
    S (ssize st1) = ssize st0.

Lemma step_count:
  ∀ n t st,
    nstar n (conf t) st →
    tsize t = n + ssize st.

Every run terminates.

Theorem steps_to_stop:
  ∀ t st,
    nstar (tsize t) (conf t) st →
    halt st.

Numbered Labeled Transitions


Inductive nlstar: nat → St → list PutPoll → St → Prop :=
| nlstar_refl: ∀ st, nlstar 0 st [] st
| nlstar_tran: ∀ n st0 e st1 tr st2,
    step st0 (Some e) st1 → nlstar n st1 tr st2 → nlstar (S n) st0 (e :: tr) st2
| nlstar_silent_tran: ∀ n st0 st1 tr st2,
    step st0 None st1 → nlstar n st1 tr st2 → nlstar (S n) st0 tr st2.
Hint Resolve nlstar_refl : core.

Lemma nlstar_transitive:
  ∀ m n st0 tr0 st1 tr1 st2,
    nlstar m st0 tr0 st1 →
    nlstar n st1 tr1 st2 →
    nlstar (m + n) st0 (tr0 ++ tr1) st2.

Lemma nlstar_lstar:
  ∀ n st0 tr st1,
    nlstar n st0 tr st1 → lstar st0 tr st1.

Lemma lstar_nlstar:
  ∀ st0 tr st1,
    lstar st0 tr st1 →
    ∃ n, nlstar n st0 tr st1.

Lemma nlstar_step_size:
  ∀ n st0 tr st1,
    nlstar n st0 tr st1 →
    ssize st1 ≤ ssize st0.

Lemma lstar_nlstar_size:
  ∀ st0 tr st1,
    lstar st0 tr st1 →
    nlstar (ssize st0 - ssize st1) st0 tr st1.

The reverse version of nlstar.

Inductive rlstar: nat → St → list PutPoll → St → Prop :=
| rlstar_refl: ∀ st, rlstar 0 st [] st
| rlstar_tran: ∀ n st0 e st1 tr st2,
    rlstar n st0 tr st1 → step st1 (Some e) st2 →
    rlstar (S n) st0 (tr ++ [e]) st2
| rlstar_silent_tran: ∀ n st0 st1 tr st2,
    rlstar n st0 tr st1 → step st1 None st2 →
    rlstar (S n) st0 tr st2.
Hint Resolve rlstar_refl : core.

Lemma rlstar_transitive:
  ∀ m n st0 tr0 st1 tr1 st2,
    rlstar m st0 tr0 st1 →
    rlstar n st1 tr1 st2 →
    rlstar (m + n) st0 (tr0 ++ tr1) st2.

Lemma rlstar_lstar:
  ∀ n st0 tr st1,
    rlstar n st0 tr st1 → lstar st0 tr st1.

Lemma lstar_rlstar:
  ∀ st0 tr st1,
    lstar st0 tr st1 →
    ∃ n, rlstar n st0 tr st1.

Lemma rlstar_nlstar:
  ∀ n st0 tr st1,
    rlstar n st0 tr st1 ↔ nlstar n st0 tr st1.
