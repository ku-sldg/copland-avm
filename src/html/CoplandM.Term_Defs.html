Terms and Evidence

A term is either an atomic ASP, a remote call, a sequence of terms with data a dependency, a sequence of terms with no data dependency, or parallel terms.
Plc represents a place.

Notation Plc := nat (only parsing).
Notation Loc := nat (only parsing).
Notation ASP_ID := nat (only parsing).
Notation N_ID := nat (only parsing).
Notation Arg := nat (only parsing).

Notation VM_ID := nat (only parsing).

Inductive ASP: Set :=
| CPY: ASP
| ASPC: ASP_ID → list Arg → ASP
| SIG: ASP
| HSH: ASP.

Inductive SP: Set :=
| ALL
| NONE.

Definition Split: Set := (SP × SP).

Inductive Term: Set :=
| asp: ASP → Term
| att: Plc → Term → Term
| lseq: Term → Term → Term
.

Definition LocRange: Set := list Loc.

Definition getTwoLocs (ls:list Loc): option (Loc×Loc) :=
 
    match ls with
    | x :: y :: ls' ⇒ ret (x,y)
    | _ ⇒ None
    end.

The structure of evidence.

Inductive Evidence: Set :=
| mt: Evidence
| uu: ASP_ID → list Arg → Plc → Evidence → Evidence
| gg: Plc → Evidence → Evidence
| hh: Plc → Evidence → Evidence
| nn: N_ID → Evidence → Evidence
| ss: Evidence → Evidence → Evidence
.

Definition splitEv_T (sp:SP) (e:Evidence) : Evidence :=
  match sp with
  | ALL ⇒ e
  | NONE ⇒ mt
  end.

Definition eval_asp t p e :=
  match t with
  | CPY ⇒ e
  | ASPC i args ⇒ uu i args p e
  | SIG ⇒ gg p e
  | HSH ⇒ hh p e
  end.

The evidence associated with a term, a place, and some initial evidence.

Fixpoint eval (t:Term) (p:Plc) (e:Evidence) : Evidence :=
  match t with
  | asp a ⇒ eval_asp a p e
  | att q t1 ⇒ eval t1 q e
  | lseq t1 t2 ⇒ eval t2 p (eval t1 p e)
  
  end.

Events

There are events for each kind of action. This includes ASP actions such as measurement or data processing. It also includes control flow actions: a split occurs when a thread of control splits, and a join occurs when two threads join. Each event is distinguished using a unique natural number.


Inductive Ev: Set :=
| copy: nat → Plc → Ev
| umeas: nat → Plc → ASP_ID → list Arg → Ev
| sign: nat → Plc → Ev
| hash: nat → Plc → Ev
| req: nat → Loc → Plc → Plc → Ev
| rpy: nat → Loc → Plc → Plc → Ev



.

Definition eq_ev_dec:
  ∀ x y: Ev, {x = y} + {x ≠ y}.

Definition pl x : Loc :=

    ∀ e0 e1, ev e0 = ev e1 → e0 = e1.

Definition asp_event i x p :=
  match x with
  | CPY ⇒ copy i p
  | ASPC id args ⇒ umeas i p id args
  | SIG ⇒ sign i p
  | HSH ⇒ hash i p
  end.

Annotated Terms

Annotated terms are used to ensure that each distinct event has a distinct natural number. To do so, each term is annotated by a pair of numbers called a range. Let (i, k) be the label for term t. The labels will be chosen to have the property such that for each event in the set of events associated with term t, its number j will be in the range i ≤ j < k.

Definition Range: Set := nat × nat.

Inductive AnnoTerm: Set :=
| aasp: Range → LocRange → ASP → AnnoTerm
| aatt: Range → LocRange → (Loc×Loc) → Plc → Plc → VM_ID → AnnoTerm → AnnoTerm
| alseq: Range → LocRange → AnnoTerm → AnnoTerm → AnnoTerm
.

Fixpoint esize t :=
  match t with
  | aasp _ _ _ ⇒ 1
  | aatt _ _ _ _ _ _ t1 ⇒ 2 + esize t1
  | alseq _ _ t1 t2 ⇒ esize t1 + esize t2
  
  end.

Definition range x :=
  match x with
  | aasp r _ _ ⇒ r
  | aatt r _ _ _ _ _ _ ⇒ r
  | alseq r _ _ _ ⇒ r
  
  end.

Definition lrange x :=
  match x with
  | aasp _ lr _ ⇒ lr
  | aatt _ lr _ _ _ _ _ ⇒ lr
  | alseq _ lr _ _ ⇒ lr
  
  end.

Fixpoint anss (t:AnnoTerm) :=
  match t with
  | aasp _ _ _ ⇒ 0
  | aatt _ _ _ _ _ _ t ⇒ 2 + anss t
  | alseq _ _ t1 t2 ⇒ anss t1 + anss t2
  
  end.

Fixpoint nss (t:Term) :=
  match t with
  | asp _ ⇒ 0
  | att _ t ⇒ nss t + 2
  | lseq t1 t2 ⇒ nss t1 + nss t2
  
  end.

Definition two_ats := (att 1 (att 2 (asp (CPY)))).

Definition three_ats := (att 1 (att 2 (att 3 (asp (CPY))))).

Definition three_ats' := (att 1 (lseq (asp CPY) (att 2 (att 3 (asp (CPY)))))).

Definition four_ats := (lseq (att 1 (lseq (asp CPY) three_ats))
                             (att 2 (att 1 (asp CPY)))).

Lemma nss_even: ∀ t n,
    nss t = n →
    Nat.even n = true.

Require Import Maps.

Definition VMID_Map := (MapC Plc VM_ID).

Definition inc_vm_id (m:VMID_Map) (p:Plc) : (VM_ID×VMID_Map) :=
  let val := map_get m p in
  match val with
  | Some v ⇒ (v,map_replace m p (v + 1))
  | None ⇒ (0,map_set m p 1)
  end.


Fixpoint anno (t: Term) (i:nat) (ls:LocRange) (m:VMID_Map) (p:Plc) : option (nat × (VMID_Map × AnnoTerm)) :=
  match t with
  | asp x ⇒ ret (S i, (m,(aasp (i, S i) [] x)))

  | att q x ⇒
    '(req_loc,rpy_loc) <- getTwoLocs ls ;;
    let '(id,newMap) := inc_vm_id m q in
    '(j,(m',a)) <- anno x (S i) (skipn 2 ls) newMap q ;;
    ret (S j, (m', aatt (i, S j) ls (req_loc,rpy_loc) p q id a) )
                
  | lseq x y ⇒
    '(j,(m',a)) <- anno x i (firstn (nss x) ls) m p ;;
    '(k,(m'',bt)) <- anno y j (skipn (nss x) ls) m' p ;;
    ret (k, (m'',alseq (i, k) ls a bt))

        
  end.

Definition anno' t i ls p := fromSome (0, ([],aasp (0,0) [] CPY)) (anno t i ls (map_set [] p 1) p).

Definition annotated x ls p :=
  snd (snd (anno' x 0 ls p)).

Fixpoint unanno a :=
  match a with
  | aasp _ _ a ⇒ asp a
  | aatt _ _ _ _ _ q t ⇒ att q (unanno t)
  | alseq _ _ a1 a2 ⇒ lseq (unanno a1) (unanno a2)
                         
  
  end.


This predicate determines if an annotated term is well formed, that is if its ranges correctly capture the relations between a term and its associated events.


Inductive well_formed_r: AnnoTerm → Prop :=
| wf_asp_r: ∀ r x ls,
    snd r = S (fst r) →
    well_formed_r (aasp r ls x)
| wf_att_r: ∀ r ls locs p q x m,
    well_formed_r x →
    S (fst r) = fst (range x) →
    snd r = S (snd (range x)) →
    
    well_formed_r (aatt r ls locs p q m x)
                  
| wf_lseq_r: ∀ r ls x y,
    well_formed_r x → well_formed_r y →
    fst r = fst (range x) →
    snd (range x) = fst (range y) →
    snd r = snd (range y) →
    well_formed_r (alseq r ls x y)
    .
Hint Constructors well_formed_r : core.

Inductive well_formed: AnnoTerm → Prop :=
| wf_asp: ∀ r x,
    snd r = S (fst r) →
    well_formed (aasp r [] x)
| wf_att: ∀ r ls locs p q x m,
    
    well_formed x →
    S (fst r) = fst (range x) →
    snd r = S (snd (range x)) →
    

    
    fst locs ≠ snd locs →
     
    
    
    In (fst locs) ls →
    In (snd locs) ls →

    
    
    
    
    
    well_formed (aatt r ls locs p q m x)
                
| wf_lseq: ∀ r ls x y,
    well_formed x → well_formed y →
    fst r = fst (range x) →
    snd (range x) = fst (range y) →
    snd r = snd (range y) →

    
    
     
    
    list_subset (lrange x) ls →
    list_subset (lrange y) ls →
    
    
     
    
    
    

    
    NoDup ((lrange x) ++ (lrange y)) →
     
    

    
    well_formed (alseq r ls x y)
      .
Hint Constructors well_formed : core.

  match goal with
  | [H : ∀ _, _, H2: Term, H3: nat |- _] ⇒ pose_new_proof (H H2 H3)
  end.

Ltac afa' :=
  match goal with
  | [H : ∀ _, _, H2: Term, H3: nat |- _] ⇒ pose_new_proof (H H2 (S H3))
  end.

Ltac afa'' :=
  match goal with
  | [H : ∀ _, _, H2: Term, H3: nat, H4:nat, H5: AnnoTerm |- _] ⇒
    pose_new_proof (H H2 (H3)(H4) H5)
  end.
