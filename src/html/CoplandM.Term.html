Fixpoint aeval t p e :=
  match t with
  | aasp _ _ x ⇒ eval (asp x) p e
  | aatt _ _ _ _ q _ x ⇒ aeval x q e
  | alseq _ _ t1 t2 ⇒ aeval t2 p (aeval t1 p e)
  
  end.


This predicate specifies when a term, a place, and some initial evidence is related to an event. In other words, it specifies the set of events associated with a term, a place, and some initial evidence.

Inductive events: AnnoTerm → Plc → Ev → Prop :=
| evtscpy:
    ∀ r lr i p,
      fst r = i →
      events (aasp r lr CPY) p (copy i p)
| evtsusm:
    ∀ i id args r lr p,
      fst r = i →
      events (aasp r lr (ASPC id args)) p (umeas i p id args)
| evtssig:
    ∀ r lr i p,
      fst r = i →
      events (aasp r lr SIG) p (sign i p)
| evtshsh:
    ∀ r lr i p,
      fst r = i →
      events (aasp r lr HSH) p (hash i p)
| evtsattreq:
    ∀ r lr q t i p p' req_loc rpy_loc m,
      fst r = i →
      events (aatt r lr (req_loc, rpy_loc) p' q m t) p (req i req_loc p q )
| evtsatt:
    ∀ r lr q t ev p p' locs m,
      events t q ev →
      events (aatt r lr locs p' q m t) p ev
| evtsattrpy:
    ∀ r lr q t i p p' req_loc rpy_loc m,
      snd r = S i →
      events (aatt r lr (req_loc, rpy_loc) p' q m t) p (rpy i rpy_loc p q)
| evtslseql:
    ∀ r lr t1 t2 ev p,
      events t1 p ev →
      events (alseq r lr t1 t2) p ev
| evtslseqr:
    ∀ r lr t1 t2 ev p,
      events t2 p ev →
      events (alseq r lr t1 t2) p ev

Lemma events_range:
  ∀ t v p,
    well_formed_r t →
    events t p v →
    fst (range t) ≤ ev v < snd (range t).

Lemma at_range:
  ∀ x r i,
    S (fst r) = fst x →
    snd r = S (snd x) →
    fst r ≤ i < snd r →
    i = fst r ∨
    fst x ≤ i < snd x ∨
    i = snd x.

Lemma lin_range:
  ∀ x y i,
    snd x = fst y →
    fst x ≤ i < snd y →
    fst x ≤ i < snd x ∨
    fst y ≤ i < snd y.

Lemma bra_range:
  ∀ x y r i,
    S (fst r) = fst x →
    snd x = fst y →
    snd r = S (snd y) →
    fst r ≤ i < snd r →
    i = fst r ∨
    fst x ≤ i < snd x ∨
    fst y ≤ i < snd y ∨
    i = snd y.

  end.

Lemma events_injective:
  ∀ t p v1 v2,
    well_formed_r t →
    events t p v1 →
    events t p v2 →
    ev v1 = ev v2 →
    v1 = v2.

Inductive evalR : Term → Plc → Evidence → Evidence → Prop :=
| evalR_asp: ∀ a p e,
    evalR (asp a) p e (eval_asp a p e)
| evalR_att: ∀ t1 q e e' p,
    evalR t1 q e e' →
    evalR (att q t1) p e e'
| evalR_lseq: ∀ t1 t2 p e e' e'',
    evalR t1 p e e' →
    evalR t2 p e' e'' →
    evalR (lseq t1 t2) p e e''
