Copland specific event systems.

Require Import Omega Preamble More_lists Term Event_system.

Construct an event system from an annotated term, place, and evidence.

Inductive PutPoll: Set :=
| putpoll: nat → nat → nat → PutPoll
| none_putpoll: nat → PutPoll.

Fixpoint store_sys (t: AnnoTerm) : EvSys PutPoll :=
  match t with
  | aasp (i, j) x ⇒ leaf (i, j) (none_putpoll i)
  | aatt (i, j) q x ⇒
    leaf (i, j) (putpoll i i j)
         
      
  | alseq r x y ⇒ before r (store_sys x)
                         (store_sys y)

                         
  | abseq r s x y ⇒ before r (store_sys x)
                         (store_sys y)

    
    

  | abpar (i, j) s x y ⇒
    merge (S i, (pred j))
          (leaf (0,0) (putpoll (fst (range x)) (fst (range x)) (snd (range x) - 1)))
          (leaf (0,0) (putpoll (fst (range y)) (fst (range y)) (snd (range y) - 1)))


   
                         








          

    
  
  end.


lseq is associative relative to the event semantics

Lemma lseq_assoc:
  ∀ t1 t2 t3 i,
    same_rel
      (store_sys (snd (anno (lseq t1 (lseq t2 t3)) i)))
      (store_sys (snd (anno (lseq (lseq t1 t2) t3) i))).
