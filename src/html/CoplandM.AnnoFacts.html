Ltac same_index :=
  match goal with
  | [H: anno ?t _ _ _ _ = Some (?n,(_, _)),
        H': anno ?t _ _ _ _ = Some (?n', (_,_)) |- _] ⇒
    assert_new_proof_by (n = n') eauto
  end.

Lemma same_anno_range: ∀ t i l l2 a b n n' p p' m m' m'' m''',
    anno t i l m p = Some (n,(m'',a)) →
    anno t i l2 m' p' = Some (n',(m''',b)) →
    n = n'.

Lemma anno_mono : ∀ (t:Term) (i j:nat) (t':AnnoTerm) (ls:LocRange) (p:Plc) m m',
  anno t i ls m p = Some (j,(m',t')) →
  j > i.
Hint Resolve anno_mono : core.

Lemma anno_range:
  ∀ x i j ls t' m m' p,
     anno x i ls m p = Some (j,(m',t')) →
    range (t') = (i, j).


Ltac do_list_empty :=
  match goal with
    [H: length ?ls = 0 |- _] ⇒
    assert_new_proof_by (ls = []) ltac:(destruct ls; solve_by_inversion)
  end.

Lemma anno_lrange:
  ∀ x i j ls t' p m m',
    length ls = nss x →
    anno x i ls m p = Some (j,(m',t')) →
    list_subset ls (lrange t').

Lemma anno_lrange'
  : ∀ (x : Term) (i j : nat) (ls : list nat)
      (t' : AnnoTerm) (p:Plc) m m',
    
    anno x i ls m p = Some (j,(m', t')) →
    list_subset (lrange t') ls.

Lemma both_anno_lrange
  : ∀ (x : Term) (i j : nat) (ls : list nat)
      (ls' : LocRange) (t' : AnnoTerm) (p:Plc) m m',
    length ls = nss x →
    anno x i ls m p = Some (j,(m',t')) →
    list_subset (lrange t') ls ∧ list_subset ls (lrange t').


Lemma nss_iff_anss: ∀ t i ls n a p m m',
    anno t i ls m p = Some (n,(m',a)) →
    nss t = anss a.

Lemma anno_some_fact: ∀ t i ls n a p m m',
    anno t i ls m p = Some (n, (m',a)) →
    length ls ≥ nss t.

Ltac do_anno_some_fact :=
  repeat
    match goal with
    | [H: anno ?t _ ?ls _ _ = Some (_,(_,_)) |- _] ⇒
      assert_new_proof_by (length ls ≥ nss t) ltac:(eapply anno_some_fact; apply H)
    end.

Ltac list_facts :=

Lemma list_too_short: ∀ t i ls p m,
      anno t i ls m p = None →
      length ls < nss t.

Lemma anno_some: ∀ t i l p m,
  length l = nss t →
  ∃ res,
    anno t i l m p = Some res.


Lemma lrange_nss: ∀ t i ls n a p m m',
    length ls = nss t →
    anno t i ls m p = Some (n,(m', a)) →
    length (lrange a) = nss t.

Lemma anno_well_formed_r:
  ∀ t i j ls t' p m m',
    
    anno t i ls m p = Some (j, (m',t')) →
    well_formed_r t'.
