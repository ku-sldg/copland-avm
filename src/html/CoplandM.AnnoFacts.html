Lemma anno_mono : ∀ (t:Term) (i j:nat) (t':AnnoTerm) (p:Plc),
  anno t i p = (j,t') →
  j > i.
Hint Resolve anno_mono : core.

Lemma anno_range:
  ∀ x i j t' p,
     anno x i p = (j,t') →
    range (t') = (i, j).


Ltac do_list_empty :=
  match goal with
    [H: length ?ls = 0 |- _] ⇒
    assert_new_proof_by (ls = []) ltac:(destruct ls; solve_by_inversion)
  end.


Lemma anno_well_formed_r:
  ∀ t i j t' p,
    
    anno t i p = (j,t') →
    well_formed_r t'.
